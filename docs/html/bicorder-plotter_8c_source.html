71 uint8_t Plot_ValueToCoord(int32_t value, int32_t y_min, int32_t y_max) {
72  int64_t new_val;
73  // Map to [0, PLOT_HEIGHT]:
74  new_val = (uint64_t) value;
75  new_val -= y_min;
76  new_val *= PLOT_HEIGHT;
77  new_val /= y_max - y_min;
78  // truncate:
79  if (new_val < 0) new_val = 0;
80  else if (new_val > PLOT_HEIGHT) new_val = PLOT_HEIGHT;
81  return (uint8_t) new_val;
82 }
