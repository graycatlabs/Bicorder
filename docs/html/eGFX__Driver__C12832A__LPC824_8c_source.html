65 void C12832A_writeCommand(C12832A_config *screen, uint8_t command) {
66  SPI_DATA_SETUP_T xfer;
67  uint16_t tx = (uint16_t) command;
68  xfer.DataSize = 8;
69  xfer.Length = 1;
70  xfer.pTx = &tx;
71  Chip_SPIM_AssertSSEL(screen->lpc_spi, screen->ssel_num);
72  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, screen->a0_pin, C12832A_A0_CMD);

112 
113  /*
114  // Initialization procedure given in datasheet:
115 
116  C12832A_writeCommand(screen, C12832A_CMD_ADC_NORMAL); // RAM address correspondence = normal
117  C12832A_writeCommand(screen, C12832A_CMD_DISPLAY_OFF); // RAM address correspondence = normal
118  C12832A_writeCommand(screen, C12832A_CMD_COM_NORMAL); // COM output scan direction = normal
119  C12832A_writeCommand(screen, C12832A_CMD_BIAS_9); // Drive voltage to 1/9 bias

121  delayms(10);
122  C12832A_writeCommand(screen, C12832A_CMD_RESISTOR_RATIO | 1); // Internal resistor ratio (Rb/Ra) to 1
123  C12832A_SetContrast(screen, 0x3f);
124  */

127  Chip_SPIM_DeAssertSSEL(screen->lpc_spi, screen->ssel_num);
128  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, screen->a0_pin, C12832A_A0_CMD);
129  delayms(100);
130 
131  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, screen->rst_pin, 0);
132  delayms(500);
133  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, screen->rst_pin, 1);
134  delayms(100);
135 
136  C12832A_writeCommand(screen, CMD_SET_DISP_START_LINE);

140  C12832A_writeCommand(screen, C12832A_CMD_BIAS_9);

143  delayms(50);
144  C12832A_writeCommand(screen, C12832A_CMD_POWER_CONTROL | 0x6);

169 extern void eGFX_Dump(eGFX_ImagePlane *Image, C12832A_config *screen) {
170  SPI_DATA_SETUP_T xfer;
171  uint32_t offset;
172  uint16_t tx[128];
173  uint8_t column, page, bit, y, mask;
174 
175  xfer.DataSize = 8;
176  xfer.pTx = tx;
177 
178  Chip_SPIM_AssertSSEL(screen->lpc_spi, screen->ssel_num);
179 
180  y = 0;
181  for (page=0; page<4; page++) {
182  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, screen->a0_pin, C12832A_A0_CMD);
183 
184  // Set page address:
185  tx[0] = (uint16_t) (C12832A_CMD_PAGE | (page & 0xf));
186 
187  // Set to first column:
188  tx[1] = (uint16_t) (C12832A_CMD_COL_LOW);
189  tx[2] = (uint16_t) C12832A_CMD_COL_HIGH;
190  xfer.Length = 3;
191  Chip_SPI_WriteFrames_Blocking(screen->lpc_spi, &xfer);
192 
193  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, screen->a0_pin, C12832A_A0_DATA); // data mode
194 
195  for (column=0; column<128; column++) {
196  tx[column] = 0;
197  for (bit=0; bit<8; bit++) {
198 
199  offset = ((y+bit) * 16) + (column>>3);
200  mask = 0x01 << (column & 0x07);
201 
202  if(Image->Data[offset] & mask) {
203  tx[column] |= 1<<bit;
204  }
205  }
206  }
207  xfer.Length = 128;
208  Chip_SPI_WriteFrames_Blocking(screen->lpc_spi, &xfer);
209  y += 8;
210  }
211 
212  Chip_SPIM_DeAssertSSEL(screen->lpc_spi, screen->ssel_num);
213 }
