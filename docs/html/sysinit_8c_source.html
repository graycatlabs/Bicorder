39 void pinMux(void) {
40  int i;
41  // Switch Matrix config:
42  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_SWM);
43 
44  Chip_SWM_MovablePinAssign(SWM_SPI0_MOSI_IO, SPI_MOSI_PIN);
45  Chip_SWM_MovablePinAssign(SWM_SPI0_SCK_IO, SPI_SCK_PIN);
46  Chip_SWM_MovablePinAssign(SWM_LCD_CS, LCD_CS_PIN);
47 
48  Chip_SWM_EnableFixedPin(SWM_FIXED_ADC8);
49 
50  Chip_SWM_FixedPinEnable(SWM_FIXED_XTALIN, 1);
51  Chip_SWM_FixedPinEnable(SWM_FIXED_XTALOUT, 1);
52 
53  Chip_SWM_EnableFixedPin(SWM_FIXED_I2C0_SDA);
54 
55 
56  // IOCON config:
57  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_IOCON);
58 
59  // Disable pullups on xtal in/out:
60  Chip_IOCON_PinSetMode(LPC_IOCON, IOCON_PIO8, PIN_MODE_INACTIVE);
61  Chip_IOCON_PinSetMode(LPC_IOCON, IOCON_PIO9, PIN_MODE_INACTIVE);
62 
63  Chip_IOCON_PinSetMode(LPC_IOCON, IOCON_PIO18, PIN_MODE_INACTIVE);
64 
65  /* Enable Fast Mode Plus for I2C pins */
66  Chip_IOCON_PinSetI2CMode(LPC_IOCON, IOCON_PIO11, PIN_I2CMODE_FASTPLUS);
67 
68 
69  Chip_IOCON_PinSetI2CMode(LPC_IOCON, IOCON_PIO10, PIN_I2CMODE_GPIO);
70 
71  Chip_GPIO_SetPinDIROutput(LPC_GPIO_PORT, 0, MoonLander_IO_SCL);
72  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, MoonLander_IO_SCL, 1);
73 
74  Chip_GPIO_Init(LPC_GPIO_PORT);
75 
76  for (i=0; i<32; i++) {
77  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, MoonLander_IO_SCL, 0);
78  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, MoonLander_IO_SCL, 1);
79  }
80 
81  Chip_IOCON_PinSetI2CMode(LPC_IOCON, IOCON_PIO10, PIN_I2CMODE_FASTPLUS);
82 
83  Chip_SWM_EnableFixedPin(SWM_FIXED_I2C0_SCL);
84 
85  Chip_Clock_DisablePeriphClock(SYSCTL_CLOCK_IOCON);
86  Chip_Clock_DisablePeriphClock(SYSCTL_CLOCK_SWM);
87 
88 }
