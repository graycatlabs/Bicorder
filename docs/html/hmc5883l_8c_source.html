36 uint8_t HMC5883L_GetXYZ(HMC5883L *hmc, int_fp *x, int_fp *y, int_fp *z) {
37  uint8_t data_buffer[6], cmd, ret;
38  int_fp divider_fp;
39  int16_t x_raw, y_raw, z_raw;
40  if (!hmc->connected) return 0;
41  cmd = HMC5883L_REG_X_MSB;
42  ret = MoonLander_I2C_Transaction(hmc->i2c, HMC5883L_I2C_ADDR,
43  &cmd, 1, data_buffer, 6);
44  if (!ret) return 0;
45 
46  x_raw = (data_buffer[0]<<8) | data_buffer[1];
47  z_raw = (data_buffer[2]<<8) | data_buffer[3];
48  y_raw = (data_buffer[4]<<8) | data_buffer[5];
49 
50  divider_fp = FP_INT(hmc->divider);
51 
52 
53  if (x_raw == -4096) (*x) = FP_INT(9000);
54  else {
55  (*x) = FP_INT((int32_t) x_raw);
56  (*x) = int_fp_div((*x), divider_fp); // In Gauss
57  (*x) = int_fp_mul((*x), FP_INT(100)); // In uTesla
58  }
59 
60  if (y_raw == -4096) (*y) = FP_INT(9000);
61  else {
62  (*y) = FP_INT((int32_t) y_raw);
63  (*y) = int_fp_div((*y), divider_fp); // In Gauss
64  (*y) = int_fp_mul((*y), FP_INT(100)); // In uTesla
65  }
66  if (z_raw == -4096) (*z) = FP_INT(9000);
67  else {
68  (*z) = FP_INT((int32_t) z_raw);
69  (*z) = int_fp_div((*z), divider_fp); // In Gauss
70  (*z) = int_fp_mul((*z), FP_INT(100)); // In uTesla
71  }
72  return 1;
73 }

116 uint8_t HMC5883L_Init(HMC5883L *hmc, LPC_I2C_T *i2c) {
117  uint8_t tx_buffer[2], ret;
118  uint8_t id_buffer[3];
119  hmc->i2c = i2c;
120  tx_buffer[0] = HMC5883L_REG_ID_A;
121  ret = MoonLander_I2C_Transaction(hmc->i2c, HMC5883L_I2C_ADDR,
122  tx_buffer, 1, id_buffer, 3);
123 
124  if (ret && id_buffer[0] == HMC5883L_ID_A &&
125  id_buffer[1] == HMC5883L_ID_B &&
126  id_buffer[2] == HMC5883L_ID_C) {
127  hmc->connected = 1;
128  }
129  else {
130  hmc->connected = 0;
131  return 0;
132  }
133 
134  if (!HMC5883L_SetRange(hmc, HMC5883L_RANGE_1_3)) return 0;
135  // Set to continuous sampling (@ default 15Hz):
136  tx_buffer[0] = HMC5883L_REG_MODE;
137  tx_buffer[1] = 0x00;
138  ret = MoonLander_I2C_Write(hmc->i2c, HMC5883L_I2C_ADDR, tx_buffer, 2);
139  //if (!ret) return 0;
140  hmc->connected = 1;
141  return 1;
142 }
