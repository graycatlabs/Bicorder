141 void adcInit(void) {
142  /* Setup ADC for 12-bit mode and normal power */
143  Chip_ADC_Init(LPC_ADC, 0);
144 
145  /* Need to do a calibration after initialization and trim */
146  Chip_ADC_StartCalibration(LPC_ADC);
147  while (!(Chip_ADC_IsCalibrationDone(LPC_ADC))) {}
148 
149  /* Setup for maximum ADC clock rate using sycnchronous clocking */
150  Chip_ADC_SetClockRate(LPC_ADC, 500000);
151 
152  Chip_ADC_SetupSequencer(LPC_ADC, ADC_SEQA_IDX,
153  (ADC_SEQ_CTRL_CHANSEL(MoonLander_IO8_ADC) | ADC_SEQ_CTRL_MODE_EOS));
154 
155  /* Setup threshold 0 low and high values to about 25% and 75% of max */
156  Chip_ADC_SetThrLowValue(LPC_ADC, 0, ((1 * 0xFFF) / 4));
157  Chip_ADC_SetThrHighValue(LPC_ADC, 0, ((3 * 0xFFF) / 4));
158 
159  /* Clear all pending interrupts */
160  Chip_ADC_ClearFlags(LPC_ADC, Chip_ADC_GetFlags(LPC_ADC));
161 
162  /* Enable ADC overrun and sequence A completion interrupts */
163  Chip_ADC_EnableInt(LPC_ADC, (ADC_INTEN_SEQA_ENABLE | ADC_INTEN_OVRRUN_ENABLE));
164 
165  /* Use threshold 0 for ADC channel and enable threshold interrupt mode for
166  channel as crossing */
167  Chip_ADC_SelectTH0Channels(LPC_ADC, ADC_THRSEL_CHAN_SEL_THR1(MoonLander_IO8_ADC));
168  Chip_ADC_SetThresholdInt(LPC_ADC, MoonLander_IO8_ADC, ADC_INTEN_THCMP_CROSSING);
169 
170  /* Enable ADC NVIC interrupt */
171  NVIC_EnableIRQ(ADC_SEQA_IRQn);
172 
173  /* Enable sequencer */
174  Chip_ADC_EnableSequencer(LPC_ADC, ADC_SEQA_IDX);
175 }

200 void SysTick_Handler(void) {
201  //static uint32_t blink_counter = 0;
202  static uint32_t sample_counter_temp = 0;
203  static uint32_t sample_counter_rh = 10;
204 
205  static uint32_t sample_counter_mag = 20;
206  static int_fp mag_x=0, mag_y=0, mag_z=0;
207 
208  static uint32_t sample_counter_range = 30;
209 
210  static uint32_t left_sw_hold_counter = 0;
211  static uint8_t left_sw_engaged = 0;
212  static uint32_t right_sw_hold_counter = 0;
213  static uint8_t right_sw_engaged = 0;
214 
215  g_delayms_counter += MS_PER_TICK;
216 
217  sample_counter_temp++;
218  if (sample_counter_temp > SAMPLE_PERIOD_TEMP) {
219  Plot_AddSample(&g_plot_temp, HTU21D_GetTemp(&g_HTU21D));
220  sample_counter_temp = 0;
221  }
222 
223  sample_counter_rh++;
224  if (sample_counter_rh > SAMPLE_PERIOD_RH) {
225 
226  Plot_AddSample(&g_plot_rh, HTU21D_GetRH(&g_HTU21D));
227  sample_counter_rh = 0;
228  }
229 
230  sample_counter_mag++;
231  if (sample_counter_mag > SAMPLE_PERIOD_MAG) {
232  HMC5883L_GetXYZ(&g_HMC5883L, &mag_x, &mag_y, &mag_z);
233  Plot_AddSample(&g_plot_mag, mag_z);
234  Compass_UpdateXY(&g_compass, mag_x, -mag_y);
235  sample_counter_mag = 0;
236  }
237 
238  sample_counter_range++;
239  if (sample_counter_range > SAMPLE_PERIOD_RANGE) {
240 
241  if (g_adc_sample_ready) {
242  g_adc_sample_ready = 0;
243  sample_counter_range = 0;
244  g_range_raw = ADC_DR_RESULT(Chip_ADC_GetDataReg(LPC_ADC, MoonLander_IO8_ADC));
245  Chip_ADC_StartSequencer(LPC_ADC, ADC_SEQA_IDX);
246  }
247  }
248 
249  if (left_sw_engaged) {
250  if (Chip_GPIO_GetPinState(LPC_GPIO_PORT, 0, SW_LEFT)) {
251  left_sw_engaged = 0;
252  left_sw_hold_counter = 0;
253  }
254  else {
255  left_sw_hold_counter += MS_PER_TICK;
256  if (left_sw_hold_counter >= SW_OFF_HOLD_TIME_MS) {
257  g_go_to_sleep = 1;
258  left_sw_hold_counter = 0;
259  left_sw_engaged = 0;
260  }
261  }
262  }
263 
264  if (right_sw_engaged) {
265  if (Chip_GPIO_GetPinState(LPC_GPIO_PORT, 0, SW_RIGHT)) {
266  right_sw_engaged = 0;
267  right_sw_hold_counter = 0;
268  }
269  else {
270  right_sw_hold_counter += MS_PER_TICK;
271  if (right_sw_hold_counter >= SW_OFF_HOLD_TIME_MS) {
272  g_go_to_sleep = 1;
273  right_sw_hold_counter = 0;
274  right_sw_engaged = 0;
275  }
276  }
277  }
278 
279  if (g_sw_left_debouncing) {
280  left_sw_engaged = 1;
281  g_sw_left_debounce_counter += MS_PER_TICK;
282  if (g_sw_left_debounce_counter >= SW_DEBOUNCE_MS) {
283  g_sw_left_debouncing = 0;
284  g_sw_left_debounce_counter = 0;
285  }
286  }
287 
288  if (g_sw_right_debouncing) {
289  right_sw_engaged = 1;
290  g_sw_right_debounce_counter += MS_PER_TICK;
291  if (g_sw_right_debounce_counter >= SW_DEBOUNCE_MS) {
292  g_sw_right_debouncing = 0;
293  g_sw_right_debounce_counter = 0;
294  }
295  }
296 
297  g_millis_counter += MS_PER_TICK;
298 }

334 void ADC_SEQA_IRQHandler(void)
335 {
336  uint32_t pending;
337 
338  /* Get pending interrupts */
339  pending = Chip_ADC_GetFlags(LPC_ADC);
340 
341  /* Sequence A completion interrupt */
342  if (pending & ADC_FLAGS_SEQA_INT_MASK) {
343  g_adc_sample_ready = 1;
344  }
345 
346  /* Clear any pending interrupts */
347  Chip_ADC_ClearFlags(LPC_ADC, pending);
348 }

353 float getRangeCentimeters(void) {
354  float v_range;
355  v_range = ADC_V_PER_LSB * g_range_raw;

357  return 61.681 * pow(v_range, -1.133);

371 void goToSleep(void) {
372  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, LCD_BACKLIGHT, 1);
373  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, RANGE_POWER_PIN, 0);
374 
375  // Give plenty of time to release switch:
376  delayms(750);
377 
378  // Disable systick timer:
379  SysTick->CTRL &= ~SysTick_CTRL_ENABLE_Msk;
380 
381  // Only allow enabled IRQ's to wake us up:
382  NVIC_DisableIRQ(ADC_SEQA_IRQn);
383  g_go_to_sleep = 0;
384  Chip_SYSCTL_EnablePINTWakeup(SW_LEFT_PININT);
385  Chip_SYSCTL_EnablePINTWakeup(SW_RIGHT_PININT);
386  SCB->SCR &= ~(SCB_SCR_SEVONPEND_Msk);
387  Chip_PMU_SleepState(LPC_PMU);
388 }

393 void wakeup(void) {
394  Chip_SYSCTL_DisablePINTWakeup(SW_LEFT_PININT);
395  Chip_SYSCTL_DisablePINTWakeup(SW_RIGHT_PININT);
396  g_go_to_sleep = 0;
397 
398  Plot_Clear(&g_plot_temp);
399  Plot_Clear(&g_plot_rh);
400 
401  Plot_Clear(&g_plot_mag);
402  Plot_Clear(&g_plot_mag);
403 
404  SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk |
405  SysTick_CTRL_TICKINT_Msk |
406  SysTick_CTRL_ENABLE_Msk;
407 
408 
409  g_sw_right_debouncing = 1;
410  g_sw_left_debouncing = 1;
411  g_sw_right_debounce_counter = 0;
412  g_sw_left_debounce_counter = 0;
413 
414 
415  NVIC_EnableIRQ(ADC_SEQA_IRQn);
416 
417  Chip_ADC_StartSequencer(LPC_ADC, ADC_SEQA_IDX);
418 
419  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, LCD_BACKLIGHT, 0);
420 
421  Chip_GPIO_SetPinState(LPC_GPIO_PORT, 0, RANGE_POWER_PIN, 1);
422 
423  Chip_PININT_ClearIntStatus(LPC_PININT, SW_LEFT_PININTCH);
424  Chip_PININT_ClearIntStatus(LPC_PININT, SW_RIGHT_PININTCH);
425 
426  NVIC_EnableIRQ(SW_LEFT_IRQn);
427  NVIC_EnableIRQ(SW_RIGHT_IRQn);
428 }
